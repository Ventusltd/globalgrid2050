<script>
function format(n){
return Math.round(n).toLocaleString();
}

function recalcAll(){
let f = parseFloat(document.getElementById("factor").value);

document.querySelectorAll("#tbl tbody tr").forEach(row=>{
let cu = parseFloat(row.dataset.cu);
let al = parseFloat(row.dataset.al);

let gbp_usd = 1.3341;
let gbp_eur = 1.1529;

let cu_net = cu / f;
let al_net = al / f;

let cu_gbp = cu_net / gbp_usd;
let al_gbp = al_net / gbp_usd;

let cu_eur = cu_gbp * gbp_eur;
let al_eur = al_gbp * gbp_eur;

row.querySelector(".cu_eur").innerText = format(cu_eur);
row.querySelector(".al_eur").innerText = format(al_eur);
row.querySelector(".cu_gbp").innerText = format(cu_gbp);
row.querySelector(".al_gbp").innerText = format(al_gbp);
});
}
</script>
